📚 Learn Error Handling Step by Step

1️⃣ TRY/CATCH - Basic Error Handling

What is try/catch? A way to handle errors gracefully instead of crashing the program

Step 1.1: Basic Try/Catch

try {
    // Code that might throw an error
    let result = riskyOperation();
    console.log("Success:", result);
} catch (error) {
    // Handle the error
    console.log("Error occurred:", error.message);
} finally {
    // Always runs
    console.log("This always executes");
}

Step 1.2: Different Types of Errors

// ReferenceError: variable doesn't exist
try {
    console.log(undefinedVariable);
} catch (error) {
    console.log("ReferenceError:", error.message);
}

// TypeError: wrong type of operation
try {
    null.toString();
} catch (error) {
    console.log("TypeError:", error.message);
}

2️⃣ THROWING CUSTOM ERRORS

What is throw? Manually create and throw your own errors

Step 2.1: Throwing Errors

function divide(a, b) {
    if (b === 0) {
        throw new Error("Cannot divide by zero!");
    }
    return a / b;
}

try {
    let result = divide(10, 0);
} catch (error) {
    console.log("Caught error:", error.message);
}

Step 2.2: Custom Error Types

class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = "ValidationError";
    }
}

function validateEmail(email) {
    if (!email.includes("@")) {
        throw new ValidationError("Invalid email format");
    }
    return true;
}

3️⃣ ASYNC ERROR HANDLING - Promises and Async/Await

How to handle errors in async code? Use try/catch with async/await or .catch() with promises

Step 3.1: Async/Await Error Handling

async function fetchUserData(userId) {
    try {
        let response = await fetch(`https://api.example.com/users/${userId}`);
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        let userData = await response.json();
        return userData;
    } catch (error) {
        console.log("Failed to fetch user data:", error.message);
        throw error; // Re-throw to let caller handle it
    }
}

Step 3.2: Promise Error Handling

function riskyAsyncOperation() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            let success = Math.random() > 0.5;
            if (success) {
                resolve("Operation successful!");
            } else {
                reject(new Error("Operation failed randomly"));
            }
        }, 1000);
    });
}

// Using .catch()
riskyAsyncOperation()
    .then(result => console.log(result))
    .catch(error => console.log("Promise error:", error.message));

4️⃣ PRACTICAL ERROR HANDLING EXAMPLES

Step 4.1: Form Validation

function validateForm(formData) {
    try {
        if (!formData.name || formData.name.trim() === "") {
            throw new Error("Name is required");
        }
        if (!formData.email || !formData.email.includes("@")) {
            throw new Error("Valid email is required");
        }
        if (!formData.age || formData.age < 18) {
            throw new Error("Must be 18 or older");
        }
        return { success: true, message: "Form is valid" };
    } catch (error) {
        return { success: false, message: error.message };
    }
}

Step 4.2: API Error Handling

async function safeApiCall(url) {
    try {
        let response = await fetch(url);
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        let data = await response.json();
        return data;
    } catch (error) {
        if (error.name === 'TypeError') {
            throw new Error("Network error - check your connection");
        }
        throw error; // Re-throw other errors
    }
}

5️⃣ ERROR HANDLING PATTERNS

Step 5.1: Graceful Degradation

function loadUserPreferences() {
    try {
        let preferences = localStorage.getItem('userPrefs');
        return JSON.parse(preferences);
    } catch (error) {
        console.log("Could not load preferences, using defaults");
        return { theme: 'light', language: 'en' };
    }
}

Step 5.2: Retry Logic

async function retryOperation(operation, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await operation();
        } catch (error) {
            if (i === maxRetries - 1) throw error;
            console.log(`Attempt ${i + 1} failed, retrying...`);
            await new Promise(resolve => setTimeout(resolve, 1000));
        }
    }
}

💡 Key Takeaways

  • try/catch: Wrap risky code and handle errors gracefully
  • throw: Create and throw custom errors
  • finally: Code that always runs, regardless of errors
  • Async errors: Use try/catch with async/await, .catch() with promises
  • Error types: ReferenceError, TypeError, custom errors
  • Graceful degradation: Provide fallbacks when errors occur
  • Retry logic: Automatically retry failed operations
