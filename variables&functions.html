📚 Learn Variables and Functions Step by Step

1️⃣ VARIABLES - Storing Data

What is a variable? A container that stores a value (like a box holding something)

Step 1.1: Declaring Variables with 'let'


let age = 25;               // Number
let isStudent = true;       // Boolean (true/false)

Step 1.2: Changing Variable Values

let score = 10;
score = 20;    // Changed the value
score = 30;    // Changed again

Step 1.3: Different Data Types

let firstName = "Ali";              // String
let price = 49.99;                  // Number (decimal)
let count = 5;                      // Number (whole)
let isActive = false;               // Boolean
let hobbies = ["coding", "gaming"]; // Array (list)
let user = {name: "Ali", age: 30};  // Object

Step 1.4: Simple Math with Variables

let a = 10;
let b = 5;
let sum = a + b;           // 15
let product = a * b;       // 50
let difference = a - b;    // 5

2️⃣ FUNCTIONS - Reusable Code Blocks

What is a function? A block of code that does a specific job. Write once, use many times!

Step 2.1: Creating Your First Function

function greet() {
    console.log("Hello, World!");
}

greet();  // Call the function

Step 2.2: Functions with Parameters (Inputs)

function greetPerson(name) {
    return "Hello, " + name + "!";
}

let message = greetPerson("Hamad");  // Pass "Hamad" as input

Step 2.3: Functions with Multiple Parameters

function add(num1, num2) {
    return num1 + num2;
}

let result = add(5, 3);  // Returns 8

Step 2.4: Function with Return Value

function multiply(a, b) {
    let result = a * b;
    return result;  // Give back the answer
}

let answer = multiply(4, 5);  // answer = 20

Step 2.5: Practical Example - Calculate Discount

function calculateDiscount(price, discountPercent) {
    let discountAmount = price * (discountPercent / 100);
    let finalPrice = price - discountAmount;
    return finalPrice;
}

let originalPrice = 100;
let finalPrice = calculateDiscount(originalPrice, 20);  // 20% off

Step 2.6: Function with Default Parameters

function introduce(name, country = "Unknown") {
    return name + " is from " + country;
}

introduce("Ali", "Pakistan");     // "Ali is from Pakistan"
introduce("Sara");                 // "Sara is from Unknown"

3️⃣ COMBINING VARIABLES & FUNCTIONS

Step 3.1: Real-World Example - Calculator

let num1 = 20;
let num2 = 15;

function calculate(a, b, operation) {
    if (operation === "add") return a + b;
    if (operation === "subtract") return a - b;
    if (operation === "multiply") return a * b;
    if (operation === "divide") return a / b;
}

let result1 = calculate(num1, num2, "add");       // 35
let result2 = calculate(num1, num2, "multiply"); // 300

Step 3.2: Working with Arrays in Functions

function sumArray(numbers) {
    let total = 0;
    for (let i = 0; i < numbers.length; i++) {
        total = total + numbers[i];
    }
    return total;
}

let scores = [85, 90, 78, 95];
let totalScore = sumArray(scores);  // 348

💡 Key Takeaways

  • Variables: Store data with names (let, const, var)
  • Functions: Reusable blocks of code that perform tasks
  • Parameters: Inputs that functions receive
  • Return: Output that functions give back
  • Call: Use a function by writing its name with ()
